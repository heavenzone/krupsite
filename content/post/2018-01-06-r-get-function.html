问题描述

  • 变量name的内容为Jordan
  • 字符串变量a的内容为name

我们要通过a获取到Jordan。

解决办法

name <- "Jordan"
a <- "name"
get(a)
## [1] "Jordan"

get应用

应用1

name <- "Jordan"
address <- "北京"
tel <- "13800138000"
number <- 23

str.seq <- c('name', 'address', 'tel', "number")

for(i in str.seq) {
    print(i)
}
## [1] "name"
## [1] "address"
## [1] "tel"
## [1] "number"
for(i in str.seq) {
    print(get(i))
}
## [1] "Jordan"
## [1] "北京"
## [1] "13800138000"
## [1] 23

应用二

library(tidyverse)
testFun <- function(df, groupby, colToSum) {
    df %>% group_by(get(groupby)) %>%
    summarise(value = sum(get(colToSum))) %>%
    return()
}
diamonds %>% testFun("cut", "price")
